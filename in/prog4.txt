val sender = Module(SlowSender)
val receiver = Module(Receiver)
sender.out <> receiver.in

module SlowSender
int i
outstream out
state 1
i = 0
goto 2
state 2
goto 3
state 3
goto 4
state 4 when out.ready()
out.write(i)
i = i + 1
goto Mux(i < 5, 2, 5)
state 5
goto 5

module Receiver
int x, j
instream in
state 1
j = 0
goto 2
state 2 when in.valid()
x = in.read()
goto 3
state 3
j = j + x
goto Mux(x < 5, 2, 4)
state 4
goto 4
